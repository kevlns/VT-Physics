PROJECT(VPLibs-data_analysis)

FIND_PACKAGE(CUDAToolkit)

FILE(GLOB_RECURSE LIB_DA_HEADER_FILES
        ${VP_LIB_ROOT}/Include/DataAnalysis/*.hpp
)
FILE(GLOB_RECURSE LIB_DA_SRC_FILES
        ${VP_LIB_ROOT}/Source/DataAnalysis/*.cpp
)

ADD_LIBRARY(vpda STATIC ${LIB_DA_HEADER_FILES} ${LIB_DA_SRC_FILES})
TARGET_LINK_LIBRARIES(vpda PRIVATE CUDA::cudart)
TARGET_INCLUDE_DIRECTORIES(vpda PUBLIC ${VP_RUNTIME_ROOT}/Include)
